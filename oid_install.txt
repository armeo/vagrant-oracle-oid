1) Install the following:
sudo rpm -Uvh http://download.fedoraproject.org/pub/epel/6/x86_64/epel-release-6-8.noarch.rpm
sudo yum install wget vim xterm unzip xorg-x11-apps xorg-x11-xauth xorg-x11-utils xorg-x11-fonts-* java-1.7.0-openjdk-devel.x86_64
sudo yum install libaio glibc compat-libstdc++-33 elfutils-libelf-devel libaio-devel libgcc libstdc++ unixODBC unixODBC-devel ksh

2) Enable the following in the sshd_config file
sudo vi /etc/ssh/sshd_config
	X11Forwarding yes
	X11DisplayOffset 10
	X11UseLocalhost yes

3) Install Oracle Database
cd /etc/yum.repos.d
sudo wget http://public-yum.oracle.com/public-yum-ol6.repo
sudo wget http://public-yum.oracle.com/RPM-GPG-KEY-oracle-ol6 -O /etc/pki/rpm-gpg/RPM-GPG-KEY-oracle
sudo yum install oracle-rdbms-server-11gR2-preinstall
sudo passwd oracle (Oracle@dm1n)
sudo vi /etc/security/limits.d/90-nproc.conf
	* - nproc 16384
sudo vi /etc/selinux/config
	SELINUX=permissive
reboot
sudo su - oracle
vi .bash_profile
	# Oracle Settings
	export TMP=/tmp
	export TMPDIR=$TMP

	export ORACLE_HOSTNAME=oid.fico.com
	export ORACLE_BASE=/u01/app/oracle
	export ORACLE_HOME=$ORACLE_BASE/product/11.2.0/dbhome_1
	export ORACLE_UNQNAME=orcl
	export ORACLE_SID=orcl
	export PATH=/usr/sbin:$PATH
	export PATH=$ORACLE_HOME/bin:$PATH

	export LD_LIBRARY_PATH=$ORACLE_HOME/lib:/lib:/usr/lib
	export CLASSPATH=$ORACLE_HOME/jlib:$ORACLE_HOME/rdbms/jlib
exit
sudo mkdir -p /u01/app/oracle/product/11.2.0/dbhome_1
sudo chown -R oracle:oinstall /u01
sudo chmod -R 755 /u01
sudo su - oracle
unzip linux.x64_11gR2_database_1of2.zip
unzip linux.x64_11gR2_database_2of2.zip
cd database
./runInstaller
